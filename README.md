```java
    /**
     * a 值参考 "杨辉三角"

     * 二阶贝泽尔曲线
     * N = 3
     * a = 1,2,1

     * c = 0,1,2
     * 公式：P0 * (1-t)^2 + 2P1 * (1-t) * t + P2 * t^2

     * N = 4
     * a = 1,3,3,1
     * b = 3,2,1,0
     * c = 0,1,2,3
     * 公式: P0 * (1-t)^3 + 3P1 * (1-t)^2 * t + 3P2 * (1-t) * t^2 + P3 * t^3
     *
     * 四阶贝泽尔曲线
     * N = 5
     * a = 1,4,6,4,1
     * b = 4,3,2,1,0
     * c = 0,1,2,3,4
     * 公式： P0 * (1-t)^4 + 4P1 * (1-t)^3 * t + 6P2 * (1-t)^2 * t^2 + 4P3 * (1-t) * t^3 + P4 * t^4
     *
     * 五阶贝泽尔曲线
     * N = 6
     * a = 1,5,10,10,5,1
     * b = 5,4,3,2,1,0
     * c = 0,1,2,3,4,5
     * 公式：P0 * (1-t)^5 + 5P1 * (1-t)^4 * t + 10P2 * (1-t)^3 * t^2 + 10P3 * (1-t)^2 * t^3 + 5P4 * (1-t) * t^4 + P5 * t^5
     *
     */
```
